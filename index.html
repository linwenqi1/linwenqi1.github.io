#include <linux/kernel.h>
#include <linux/init.h>
#include <asm/io.h>

static int __init simple_module_init(void)
{
        printk("Helloworld.\n");
        return 0;
}
/*
 *The cleanup function of the module.
 */
static void __exit simple_module_exit(void)
{
        printk(KERN_INFO"Exit.\n");
}
